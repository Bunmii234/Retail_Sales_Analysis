The SQL Logic


SELECT 
    CASE 
        WHEN Age <= 24 THEN '18-24'
        WHEN Age <= 40 THEN '25-40'
        WHEN Age <= 60 THEN '41-60'
        ELSE '60+' 
    END AS Age_Group,
    Product_Category,
    SUM(Total_Amount) AS Revenue
FROM retail_sales_dataset
GROUP BY 1, 2;
